# Configuration du simulateur Artouste (format "clé valeur", éditable à la main).
# Une clé et sa valeur par ligne ; une ligne commençant par # est un commentaire.
# Toute clé absente garde sa valeur par défaut.

# terrain : sous-dossier de assets/terrain/ à charger au lancement.
# Terrains disponibles :
#   ossau        vallée d'Ossau (lac d'Artouste, pic du Midi d'Ossau), montagne
#   cote-landes  côte basco-landaise (Bayonne -> Vieux-Boucau), bord de mer
#   arcachon     bassin d'Arcachon (Cap Ferret -> Marcheprime, Biscarrosse -> Arès), bord de mer
#   cauterets    Cauterets - Gavarnie (chemin des cascades, Pont d'Espagne, cirque de Gavarnie), montagne
#   bordeaux     Bordeaux (la Garonne, port de la Lune, Mérignac, Pessac), terre intérieure
# La variable d'environnement ARTOUSTE_TERRAIN, si définie, a la priorité.
terrain ossau

# turbine_demarree : 1 = turbine et rotor d'emblée au régime au lancement (au lieu
# de la séquence de démarrage d'environ une minute), pratique pour décoller tout de
# suite en test ; 0 = démarrage normal à froid. La variable d'environnement
# ARTOUSTE_TURBINE_DEMARREE, si définie, a la priorité.
turbine_demarree 0

# demo : 1 = mode démonstration automatique au lancement. L'appareil joue tout seul
# un vol (démarrage rapide, décollage, montée et vol jusqu'à la Dune du Pilat survolée
# vers 1500 m, demi-tour et retour se poser sur le pad de départ), rejoué en boucle ;
# le terrain est alors forcé sur arcachon. Une entrée du pilote ou la touche V coupe
# la démo. La variable d'environnement ARTOUSTE_DEMO, si définie, a la priorité.
# 0 = pas de démo.
demo 0

# radio_url : URL d'un flux radio internet (MP3 sur HTTP) joué dans le cockpit.
# Touche K en vol libre pour l'allumer ou la couper. Vide = pas de radio.
# La variable d'environnement ARTOUSTE_RADIO_URL, si définie, a la priorité.
# Exemple (décommenter pour activer) :
radio_url https://stream2.madein80.live/listen/madein80/playhd.mp3

# sun_time_scale : vitesse d'écoulement du temps pour le cycle jour/nuit (course du
# soleil). Durée réelle d'une journée complète = 86400 / sun_time_scale secondes.
#   1   = temps réel : départ à l'heure locale du PC
#   72  = journée complète en 20 min (défaut)
#   144 = journée complète en 10 min
#   0   = temps figé à midi (le soleil ne bouge pas)
# En temps réel (1) le soleil part de l'heure locale du PC ; pour toute autre valeur il
# part de midi, pour démarrer sur une belle lumière. Avec 0, il reste figé à midi.
sun_time_scale 72
