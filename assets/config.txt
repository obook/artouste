# Configuration du simulateur Artouste (format "cle valeur", editable a la main).
# Une cle et sa valeur par ligne ; une ligne commencant par # est un commentaire.
# Toute cle absente garde sa valeur par defaut.

# terrain : sous-dossier de assets/terrain/ a charger au lancement.
# Terrains disponibles :
#   ossau        vallee d'Ossau (lac d'Artouste, pic du Midi d'Ossau), montagne
#   cote-landes  cote basco-landaise (Bayonne -> Vieux-Boucau), bord de mer
#   arcachon     bassin d'Arcachon (Cap Ferret -> Marcheprime, Biscarrosse -> Arès), bord de mer
#   cauterets    Cauterets - Gavarnie (chemin des cascades, Pont d'Espagne, cirque de Gavarnie), montagne
# La variable d'environnement ARTOUSTE_TERRAIN, si definie, a la priorite.
terrain ossau

# turbine_demarree : 1 = turbine et rotor d'emblee au regime au lancement (au lieu
# de la sequence de demarrage d'environ une minute), pratique pour decoller tout de
# suite en test ; 0 = demarrage normal a froid. La variable d'environnement
# ARTOUSTE_TURBINE_DEMARREE, si definie, a la priorite.
turbine_demarree 0

# demo : 1 = mode demonstration automatique au lancement. L'appareil joue tout seul
# un vol (demarrage rapide, decollage, montee et vol jusqu'a la Dune du Pilat survolee
# vers 1500 m, demi-tour et retour se poser sur le pad de depart), rejoue en boucle ;
# le terrain est alors force sur arcachon. Une entree du pilote ou la touche V coupe
# la demo. La variable d'environnement ARTOUSTE_DEMO, si definie, a la priorite.
# 0 = pas de demo.
demo 0

# radio_url : URL d'un flux radio internet (MP3 sur HTTP) joue dans le cockpit.
# Touche K en vol libre pour l'allumer ou la couper. Vide = pas de radio.
# La variable d'environnement ARTOUSTE_RADIO_URL, si definie, a la priorite.
# Exemple (decommenter pour activer) :
